#include <alloca.h>
#include <unistd.h>

int main(int argc, char *argv[], char *envp[])
{
    int ret, i;
    extern char *optarg;
    extern int optind;
    char **targv = alloca(sizeof(char *) * argc);
    for (i = 0; i < argc; ++i)
    {
        targv[i] = argv[i];
    }
    while ((ret = getopt(argc, targv, ":v:")) != -1)
    {
        // ...
    }
    optind = 1;
    for (i = 0; i < argc; ++i)
    {
        targv[i] = argv[i];
    }
    while ((ret = getopt(argc, targv, ":A:B:")) != -1)
    {
        // ...
    }
}
